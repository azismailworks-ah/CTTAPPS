function refreshFSORangkingCache(){

  const data =
  buildFSORangking();

  const sh =
  getSheet(
    'FSORangkingCache'
  );

  sh.clearContents();

  const rows =
  data.map(

    function(item){

      return [

        item.fullname,
        item.ttHandling,
        item.updates

      ];

    }

  );

  if(rows.length){

    sh.getRange(

      1,
      1,
      rows.length,
      3

    )
    .setValues(
      rows
    );

  }

}
